== PROJECT: ECOMMERCE CLI
== DESC: -

== DATA
PRODUCTS (LIST OF DICTIONARY)
    [
        { "CODE","PRODUCT_NAME","STOCK" },
        { "CODE","PRODUCT_NAME","STOCK" },
        { "CODE","PRODUCT_NAME","STOCK" }
    ]
NEW_PRODUCT (DICTIONARY)
    { "CODE","PRODUCT_NAME","STOCK" }
MONEY (INTEGER)
CART (LIST OF DICTIONARY)
    [
        { "CODE","PRODUCT_NAME","STOCK" },
        { "CODE","PRODUCT_NAME","STOCK" },
        { "CODE","PRODUCT_NAME","STOCK" }
    ]


== FUNCTIONS
MAIN_MENU()
SHOW_PRODUCTS()
ADD_NEW_PRODUCT()
REMOVE_PRODUCT()
ADD_TO_CART()
CHECK_MY_CART()
CHECKOUT()
REMOVE_PRODUCT_ON_CART()
CONFIRM_CHECKOUT()

MAIN_MENU()
- SHOW MENUS TO INTERACT WITH APP
    - [1] SHOW_PRODUCTS()
    - [2] CHECK_MY_CART()
    - [3] CHECKOUT()
    - [4] ADD_NEW_PRODUCT()
    - [5] REMOVE_PRODUCT()
    - [0] EXIT

SHOW_PRODUCTS()
- SHOW LIST OF PRODUCTS
    - [PRODUCT_CODE] ADD_TO_CART(PRODUCT_CODE)
    - [0] MAIN_MENU()

ADD_NEW_PRODUCT()
- ADD NEW PRODUCT TO PRODUCT CATALOGUE
    - INSERT PRODUCT CODE
    - INSERT PRODUCT NAME
    - INSERT PRODUCT STOCK
- BACK TO MAIN_MENU()

REMOVE_PRODUCT()
- REMOVE PRODUCT FROM PRODUCT CATALOGUE
    - IF FOUND, THEN REMOVE THE ENTIRE STOCK OF THE PRODUCT
    - IF NOT FOUND, SHOW MESSAGE PRODUCT NOT FOUND
    - ELSE, SHOW ERROR MESSAGE
- BACK TO MAIN_MENU()

ADD_TO_CART(PRODUCT_CODE)
- ADD PRODUCT TO CART
    - IF FOUND, THEN ADD 1 STOCK OF THE PRODUCT
    - IF NOT FOUND, SHOW MESSAGE PRODUCT NOT FOUND
    - ELSE, SHOW ERROR MESSAGE
- BACK TO MAIN_MENU()

CHECK_MY_CART()
- SHOW LIST OF PRODUCTS IN MY CART
- BACK TO MAIN_MENU()

CHECKOUT()
- SHOW LIST OF PRODUCTS ON MY CART TO CHECK OUT PROCESS
    - [PRODUCT_CODE] REMOVE_PRODUCT_ON_CART(PRODUCT_CODE)
    - [1] CONFIRM_CHECKOUT()
    - [0] BACK TO MAIN_MENU()

REMOVE_PRODUCT_ON_CART()
- REMOVE PRODUCT ON CART
    - IF FOUND, THEN REMOVE 1 STOCK OF PRODUCT
    - IF NOT FOUND, SHOW MESSAGE PRODUCT NOT FOUND
    - ELSE, SHOW ERROR MESSAGE
- BACK TO CHECKOUT()

CONFIRM_CHECKOUT()
- PAYMENT PROCESS, INSERT AMOUNT OF MONEY IN RUPIAH
    - INSERT AMOUNT OF MONEY
    - CONDITIONAL CASE
        - IF CORRECT AMOUNT, THEN PAYMENT SUCCESS AND BACK TO MAIN_MENU()
        - IF MONEY NOT ENOUGH, THEN ASKED AGAIN TO INSERT CORRECT AMOUNT OF MONEY
        - IF MONEY MORE THAN ASKED, THEN RETURN THE CHANGE AND BACK TO MAIN_MENU()
